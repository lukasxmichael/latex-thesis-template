\usepackage{caption}

  format=hang,
  font={normalsize},
  labelfont={bf},
  labelsep=period,
  justification=justified,
  singlelinecheck=false
}

\captionsetup[table]{
  belowskip=12pt,
  aboveskip=4pt,
  position=top
}

% Specific settings for figures  
\captionsetup[figure]{
  belowskip=4pt,
  aboveskip=12pt,
  position=bottom
}
