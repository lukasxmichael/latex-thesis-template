% True APA 7th Edition Bibliography Setup
\usepackage[
    backend=biber,
    style=apa,
    citestyle=apa,
    sortcites=true,
    sorting=nyt,
    hyperref=true,
    backref=false,
    natbib=false
]{biblatex}

% Language mapping for APA style
\DeclareLanguageMapping{ngerman}{ngerman-apa}
% For English: \DeclareLanguageMapping{american}{american-apa}

% URL breaking settings
\setcounter{biburllcpenalty}{7000}
\setcounter{biburlucpenalty}{8000}

% APA hanging indent (0.5 inch standard)
\setlength{\bibhang}{0.5in}

% Reduce space between entries (optional)
\setlength{\bibitemsep}{6pt}

\addbibresource{sources/literature.bib}
